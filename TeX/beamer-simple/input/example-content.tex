\begin{frame}
\frametitle{Theorem}
\begin{theorem}[Mass--energy equivalence]
  For mass $m$, speed of light $c$ = $299,792,458$ m/s$^2$, the energy
  equivalence is given by $E = mc^2.$
\end{theorem}
